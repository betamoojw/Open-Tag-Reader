           function updateBatchNumbers() {
               batchNumberSelect.innerHTML = '<option value="">Select Batch</option>';
               const selectedProduct = productNameSelect.value;
               if (selectedProduct) {
                   const matchingChemicals = chemicalInventory.filter(chemical => chemical.productName === selectedProduct);
                   matchingChemicals.forEach(chemical => {
                       const option = document.createElement('option');
                       option.value = chemical.batchNumber;
                       option.textContent = chemical.batchNumber;
                       batchNumberSelect.appendChild(option);
                   });
               }
               whpInput.value = '';
               esiInput.value = '';
           }

           function updateWHP() {
               const selectedProduct = productNameSelect.value;
               const selectedBatch = batchNumberSelect.value;
               if (selectedProduct && selectedBatch) {
                   const chemical = chemicalInventory.find(chem => chem.productName === selectedProduct && chem.batchNumber === selectedBatch);
                   if (chemical) {
                       whpInput.value = chemical.whp;
                       esiInput.value = chemical.whp;
                   }
               }
           }

           function getNextTreatmentIndex() {
               return treatmentCounter++;

           function displayTreatments() {

                   const actionsCell = row.insertCell();
                   const editButton = document.createElement('button');
                   editButton.textContent = 'Edit';
                   editButton.onclick = () => editTreatment(treatment);

                   const deleteButton = document.createElement('button');
                   deleteButton.textContent = 'Delete';
                   deleteButton.onclick = () => deleteTreatment(treatment.treatmentIndex);
                   actionsCell.appendChild(deleteButton);

               });
               // Scroll to the bottom of the table
               treatmentTableBody.parentElement.scrollTop = treatmentTableBody.parentElement.scrollHeight;
           }

           function openAddTreatmentForm() {
               addTreatmentForm.classList.remove('hidden');
               treatmentForm.reset();
               treatmentIndexInput.value = '';
               populateProductDropdown();
           }

           function closeAddTreatmentForm() {
               addTreatmentForm.classList.add('hidden');
           }

           function editTreatment(treatment) {
               openAddTreatmentForm();
               populateProductDropdown();
               productNameSelect.value = treatment.productName;
               updateBatchNumbers();
               batchNumberSelect.value = treatment.batchNumber;

               for (const key in treatment) {
                   if (treatmentForm.elements[key]) {
                       treatmentForm.elements[key].value = treatment[key];
                   }
               }
           }

           function deleteTreatment(index) {
               if (confirm('Are you sure you want to delete this treatment record?')) {
                   treatments = treatments.filter(treatment => treatment.treatmentIndex !== index);
                   displayTreatments();
               }
           }

           treatmentForm.addEventListener('submit', function(event) {
               event.preventDefault();
               const formData = new FormData(treatmentForm);
               const data = Object.fromEntries(formData.entries());

               if (!data.treatmentIndex) {
                   data.treatmentIndex = getNextTreatmentIndex().toString();
               }

               const existingIndex = treatments.findIndex(t => t.treatmentIndex === data.treatmentIndex);
               if (existingIndex > -1) {
                   treatments[existingIndex] = data;
               } else {
                   treatments.push(data);
               }

               displayTreatments();
               closeAddTreatmentForm();
           });
